FONKSİYON SecimAl()
    DÖNGÜ sonsuza kadar
        kullanıcıdan "Pi için 1, Fibonacci için 2 seçin" girişi al
        EĞER giriş "1" veya "2" ise

            "Hata: Lütfen 1 veya 2'yi seçin." yazdır

    DÖNGÜ BİTİR

FONKSİYON NAlVeDogrula()
    DÖNGÜ sonsuza kadar
        kullanıcıdan "N değerini girin (0-7): " girişi al

        EĞER giriş tamsayıya çevrilemiyorsa
            "Hata: N bir tamsayı olmalı." yazdır
            DEVAM ET
        EĞER BİTİR

        N = tamsayı(giriş)

        EĞER N < 0 ise
            "Hata: N negatif olamaz." yazdır
            DEVAM ET
        EĞER BİTİR

        EĞER N > 7 ise
            "Lütfen {N} sayısını azaltınız." yazdır
            PROGRAMI BİTİR
        EĞER BİTİR

        N'i DÖNDÜR
    DÖNGÜ BİTİR
FONKSİYON BİTİR


FONKSİYON PiYazdir(N)
    Pi sayısını N ondalık basamak ile formatla
    "π (N basamak): {pi}" yazdır
FONKSİYON BİTİR


FONKSİYON FibonacciTopla(N)
    a = 0
    b = 1
    toplam = 0

    a <= N olduğu sürece
        toplam = toplam + a
        (a, b) = (b, a + b)   // Python mantığı: tuple assignment
    DÖNGÜ BİTİR

    toplamı DÖNDÜR
FONKSİYON BİTİR


ANA PROGRAM (Main)
    secim = SecimAl()
    N = NAlVeDogrula()

    EĞER secim == 1 ise
        PiYazdir(N)
    DEĞİLSE
        toplam = FibonacciTopla(N)
        "N için (<=N) Fibonacci toplamı: {toplam}" yazdır
    EĞER BİTİR
