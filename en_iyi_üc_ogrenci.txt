PSEUDO KOD: For Loop ile Not Analizi + Harf Notu + Top-3 + Özet

BAŞLA

students listesi oluştur
    her eleman: {"name": <isim>, "scores": [puan1, puan2, puan3]}

FONKSİYON letter_grade(avg)
    EĞER avg >= 90 ise DÖNDÜR "A"
    DEĞİLSE EĞER avg >= 80 ise DÖNDÜR "B"
    DEĞİLSE EĞER avg >= 70 ise DÖNDÜR "C"
    DEĞİLSE EĞER avg >= 60 ise DÖNDÜR "D"
    DEĞİLSE DÖNDÜR "F"
    EĞER BİTİR
FONKSİYON BİTİR


results = boş liste
total_sum = 0
total_count = 0
best_avg = -1
best_name = boş

# Ana döngü: öğrencileri dolaş
HER öğrenci s İÇİN students içinde
    name = s["name"]
    scores = s["scores"]

    # İç içe döngü: puanları topla
    s_sum = 0
    HER puan sc İÇİN scores içinde
        s_sum = s_sum + sc
    DÖNGÜ BİTİR

    avg = s_sum / scores uzunluğu

    grade = letter_grade(avg)

    results listesine (name, avg, grade) ekle

    total_sum = total_sum + s_sum
    total_count = total_count + scores uzunluğu

    EĞER avg > best_avg ise
        best_avg = avg
        best_name = name
    EĞER BİTİR
DÖNGÜ BİTİR

class_avg = total_sum / total_count


# Top-3 için sıralama
sorted_results = results'u avg değerine göre azalan sırala
top_n = 3
top3 = boş liste

i = 0'dan min(top_n, sorted_results uzunluğu) - 1'e kadar
    top3 listesine sorted_results[i] ekle
DÖNGÜ BİTİR


# 80+ filtreleme
above_80 = boş liste
HER (name, avg, grade) İÇİN results içinde
    EĞER avg >= 80 ise
        above_80 listesine name ekle
    EĞER BİTİR
DÖNGÜ BİTİR


# Rapor yazdırma
"=== Öğrenci Raporu ===" yazdır
HER (name, avg, grade) İÇİN results içinde
    "{name} | Ortalama: {avg} | Harf: {grade}" yazdır
DÖNGÜ BİTİR

"=== Sınıf Özeti ===" yazdır
"Sınıf ortalaması: {class_avg}" yazdır
"En iyi öğrenci: {best_name} ({best_avg})" yazdır

"=== Top-3 ===" yazdır
rank = 1
HER (name, avg, grade) İÇİN top3 içinde
    "{rank}. {name} - {avg} ({grade})" yazdır
    rank = rank + 1
DÖNGÜ BİTİR

"=== 80+ Ortalama Olanlar ===" yazdır
EĞER above_80 boş değilse
    isimleri virgülle birleştirip yazdır
DEĞİLSE
    "Yok" yazdır
EĞER BİTİR

BİTİR
